\subsection{RMSE}

\setbeamertemplate{footline}[shrunklight]
% full headline for sections in headline
\setbeamertemplate{frametitle}[full]
\begin{frame}[t]
	\frametitle{RMSE$^{\dagger}$}
	%======================================== content =====%
	\begin{equation*} 
		\rmse^{\dagger} = \frac{\left \| \alpha \cdot \hat{\bm{C}} - \bm{C} \right 		\|_{F}}{\left \| \bm{C} \right \|_{F}}
	\end{equation*}\\
	
	\begin{equation*} 
		\alpha = \frac{ \vectorize(\bm{C})^{\T} \cdot \vectorize(\hat{\bm{C}}) }{ \vectorize(\hat{\bm{C}})^{\T} \cdot \vectorize(\hat{\bm{C}})} .
	\end{equation*}
	
	\centering
	where \\
	$\bm{C} =$ reference data\\ 
	$\hat{\bm{C}} =$ obtained results to compare
	%===================================================%
\end{frame}
